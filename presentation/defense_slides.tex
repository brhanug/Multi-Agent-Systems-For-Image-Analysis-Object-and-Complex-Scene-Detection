% --- SLIDE 5: FORMAL DERIVATION: SCENE-AWARE AGREEMENT ---
\begin{frame}{Formal Derivation: Scene-Aware Agreement (SAA)}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{Proposition 1: Epistemic Variance Proxy}
        Let $s_i = P(Y=1 | X, \theta_i)$. If agents are approximately independent ($r=0.57$, $\text{MI}=0.334$), the sample variance approximates epistemic uncertainty:
        $$\hat{\sigma}^2 = \frac{1}{n-1}\sum_{i=1}^{n}(s_i - \bar{s})^2 \approx \mathbb{V}_{\theta}[\mathbb{E}[Y|X,\theta]]$$
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Theorem 1: Bayesian Risk Bounds}
        Expected target risk $\mathcal{R}_T(h_{\text{AWLF}})$ under domain shift is bounded by the weighted sum of source risks, $\mathcal{H}\Delta\mathcal{H}$-divergence, and optimal joint error:
        $$\mathcal{R}_T(h_{\text{AWLF}}) \le \sum_{i=1}^n w_i \mathcal{R}_S(s_i) + \frac{1}{2} d_{\mathcal{H}\Delta\mathcal{H}}(P_S, P_T) + \lambda^*$$
        AWLF minimizes this bound by dynamically penalizing out-of-domain failure modes.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% --- SLIDE 6: SELF-TRAINING & CONVERGENCE GUARANTEES ---
\begin{frame}{Self-Training and Convergence Guarantees}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{Theorem 2: Fixed-Point Convergence}
        Let $T: \mathbf{L}^{(t)} \to \mathbf{L}^{(t+1)}$ be the SAA consensus operator mapping label matrices. If the consensus operator is contractive ($\gamma \in (0, 1)$):
        $$\|T(\mathbf{A}) - T(\mathbf{B})\| \le \gamma \|\mathbf{A} - \mathbf{B}\|$$
        then the self-training pseudo-label refinement loop converges to a unique fixed-point:
        $$\lim_{t \to \infty} \mathbf{L}^{(t)} = \mathbf{L}^* = T(\mathbf{L}^*)$$
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Structural Stability}
        \begin{itemize}
          \item Proves that SAA consensus acts as a spatial-semantic smoothing kernel.
          \item Guarantees that pseudo-label refinement does not diverge, oscillate, or accumulate out-of-domain errors over training iterations.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% --- SLIDE 9: SPATIAL AUDIT & COHEN'S KAPPA ---
\begin{frame}{Spatial Domain Gap and Inter-Annotator Agreement}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{The 200-Image Audit Results}
        \begin{itemize}
          \item Audited 200 images using a second human expert in Label Studio.
          \item YOLOv11m collapses to a real human mAP50 of **0.1316** and mAP50-95 of **0.0570** (a catastrophic **-0.8574** decay).
          \item Inter-annotator agreement yields a Cohen's Kappa score of **0.1555** overall.
          \item Category-specific binary Kappa for *Drawing* collapses to **0.0415**.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Defending Slight Agreement ($\kappa = 0.1555$)}
        \begin{itemize}
          \item Low Kappa is not a human failure; it is empirical proof of **inherent historical ambiguity** in visual culture.
          \item Proves that a monolithic ground-truth is methodologically flawed on cultural heritage archives.
          \item Consensus scoring is structurally necessary to map and preserve these epistemic gaps.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
